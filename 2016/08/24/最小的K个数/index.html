最小的K个数

题目描述

输入n个整数，找出其中最小的K个数。例如输入4,5,1,6,2,7,3,8这8个数字，则最小的4个数字是1,2,3,4,。

这样的题目大致有三个思路

*基于堆排序算法，构建最大堆。时间复杂度为O(nlogk)
*如果用快速排序，时间复杂度为O(nlogn)；
*如果用冒泡排序或者简单选择排序，时间复杂度为O(n*k)
*/

方法一 ： 使用TreeSet作为最大堆

public static ArrayList<Integer> GetLeastNumbers_Solution(int [] input, int k) {
            ArrayList<Integer> list = new ArrayList<>();
            if(input==null || input.length==0 ||k>input.length || k<=0){
                return list;
            }
            TreeSet<Integer> set = new TreeSet<>(new Comparator<Integer>() {
                @Override
                public int compare(Integer o1, Integer o2) {
                    return o2.compareTo(o1);
                }
            });
            for(int i=0;i<input.length;i++){
                while(set.size()!=k){
                    set.add(input[i++]);
                }
                if(i<input.length && input[i]<set.first()){
                    set.pollFirst();
                    set.add(input[i]);
                }
            }
            Iterator<Integer> it = set.iterator();
            while(it.hasNext()){
                list.add(it.next());
            }
            return list;
}

方法二 ： 自己构建一个最大堆 做辅助运算

public static ArrayList<Integer> GetLeastNumbers_Solution(int [] input, int k) {
        ArrayList<Integer> list = new ArrayList<>();
        if(input==null || input.length==0 ||k>input.length || k<=0){
            return list;
        }
        int N = input.length;  
        for(int i=k/2-1;i>=0;i--){  //因为从下标0开始，所以i起始值为 N/2-1,如果从下标1开始构造堆，则i起始为 N/2
            sink(input, i, k-1);
        }
        for(int i=k;i<N;i++){
            if(input[i]<input[0]){
                input[0] = input[i];
                sink(input, 0, k-1);
            }
        }
        while(k>0){
            list.add(input[--k]);
        }
        return list;
}

private static void sink(int[] input,int k,int N){
    while(2*k+1<=N){
        int j = 2*k+1;
        if(j<N && input[j]<input[j+1]){
            j++;
        }
        if(input[k]<input[j]){
            int temp = input[k];
            input[k] = input[j];
            input[j] = temp;
            k = j;
        }else{
            break;
        }
    }
}

方法三： 基于快排的切分思想，时间复杂度为 O(nlogn)

public static ArrayList<Integer> GetLeastNumbers_Solution(int [] input, int k) {
        ArrayList<Integer> list = new ArrayList<>();
        if(input==null || input.length==0 ||k>input.length || k<=0){
            return list;
        }
        int len = input.length;
        int low = 0;
        int high = len-1;
        int index = patition(input,len,low,high);
        while(index!=k-1){
            if(index < k-1){
                 low = index+1;
                 index = patition(input, len, low, high);
            }else{
                 high = index-1;
                 index = patition(input, len, low, high);
            }
        }
        for(int i=0;i<k;i++){
            list.add(input[i]);
        }
        return list;
}

private static int patition(int[] input, int len, int low, int high) {
    if(low==high){   //防止数组越界
        return low;
    }
    int p = input[low];
    int i = low;
    int j = high+1;
    while(true){
        while(input[++i]<p){
            if(i==high){
              break;
            }
        }
        while(input[--j]>p){
            if(j==low){
                break;
            }
        }
        if(i>=j){
            break;
        }
        exch(input,i,j);
    }
    exch(input,low,j);
    return j;
}

private static void exch(int[] input, int i, int j) {
    int temp = input[i];
    input[i] = input[j];
    input[j] = temp;
}

方法四： 选择排序的方法 ，时间复杂度为 O(n*k)

public static ArrayList<Integer> GetLeastNumbers_Solution(int[] input, int k) {
        int len = input.length;
        ArrayList<Integer> list = new ArrayList<>();
        if (input != null && len != 0 && k > 0 && k <= len) {
            for (int i = 0; i < k; i++) {
                int min = i;
                for (int j = i+1; j < len; j++) {
                    if (input[j] < input[min]) {
                        min = j;
                    }
                }
                if(i!=min){
                    exch(input,i,min);
                }
                list.add(input[i]);
            }
        }
        return list;
}

private static void exch(int[] input, int i, int j) {
        int temp = input[i];
        input[i] = input[j];
        input[j] = temp;
}
